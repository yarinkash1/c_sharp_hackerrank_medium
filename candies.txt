using System;
using System.Collections.Generic;
using System.Linq;

class Result
{
    /*
     * Complete the 'candies' function below.
     *
     * The function is expected to return a LONG_INTEGER.
     * The function accepts following parameters:
     * 1. INTEGER n
     * 2. INTEGER_ARRAY arr
     */

    public static long candies(int n, List<int> arr)
    {
        if (n == 0) return 0;
        
        // Use an array of long for candy counts to match the required return type (LONG_INTEGER)
        // and prevent potential overflow on the total sum.
        long[] candyCounts = new long[n];
        
        // 1. Initialize all students with the absolute minimum of 1 candy.
        for (int i = 0; i < n; i++)
        {
            candyCounts[i] = 1;
        }
        
        // --- 2. Left-to-Right Pass: Satisfy the Left Neighbor Constraint ---
        // If the current student (i) has a higher rating than their left neighbor (i-1), 
        // they must get 1 more candy than the left neighbor.
        for (int i = 1; i < n; i++)
        {
            if (arr[i] > arr[i - 1])
            {
                candyCounts[i] = candyCounts[i - 1] + 1;
            }
        }
        
        // --- 3. Right-to-Left Pass: Satisfy the Right Neighbor Constraint ---
        // If the current student (i) has a higher rating than their right neighbor (i+1),
        // they must get 1 more candy than the right neighbor.
        // We use Math.Max to ensure we keep the larger value already set by the L-to-R pass.
        for (int i = n - 2; i >= 0; i--)
        {
            if (arr[i] > arr[i + 1])
            {
                // Update only if the right constraint requires a higher candy count than already assigned
                candyCounts[i] = Math.Max(candyCounts[i], candyCounts[i + 1] + 1);
            }
        }
        
        // --- 4. Calculate Total Candies ---
        // Summing the final minimal assignments.
        long total_candies = 0;
        foreach (long candy in candyCounts)
        {
            total_candies += candy;
        }
        
        // A simpler way to sum the array:
        // long total_candies = candyCounts.Sum(); 

        return total_candies;
    }
}

// The Solution class remains unchanged, as its input/output logic was correct.
class Solution
{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable("OUTPUT_PATH"), true);

        int n = Convert.ToInt32(Console.ReadLine().Trim());

        List<int> arr = new List<int>();

        for (int i = 0; i < n; i++)
        {
            int arrItem = Convert.ToInt32(Console.ReadLine().Trim());
            arr.Add(arrItem);
        }

        long result = Result.candies(n, arr);

        textWriter.WriteLine(result);

        textWriter.Flush();
        textWriter.Close();
    }
}
